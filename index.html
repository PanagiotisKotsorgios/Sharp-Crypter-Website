EncryptionExample.cs 

using System;
using System.Security.Cryptography;
using System.IO;
using System.Collections.Generic;

namespace SharpCrypter.Core {
    public class SecureFileEncryptor {
        
        // Main encryption method
        public void EncryptFile(string inputPath, string outputPath) {
            using (var aes = Aes.Create()) {
                aes.KeySize = 256;
                aes.GenerateIV();
                
                // Generate secure key
                var key = GenerateSecureKey(aes.KeySize);
                
                using (var encryptor = aes.CreateEncryptor(key, aes.IV))
                using (var fsOutput = new FileStream(outputPath, FileMode.Create)) {
                    // Write IV
                    fsOutput.Write(aes.IV, 0, aes.IV.Length);
                    
                    using (var cryptoStream = new CryptoStream(fsOutput, encryptor, CryptoStreamMode.Write))
                    using (var fsInput = new FileStream(inputPath, FileMode.Open)) {
                        fsInput.CopyTo(cryptoStream);
                    }
                }
                
                SecureMemoryWipe(key);
            }
        }
        
        private byte[] GenerateSecureKey(int keySize) {
            using (var rng = new RNGCryptoServiceProvider()) {
                var key = new byte[keySize / 8];
                rng.GetBytes(key);
                return key;
            }
        }
        
        private void SecureMemoryWipe(byte[] data) {
            for (int i = 0; i < data.Length; i++) {
                data[i] = 0;
            }
        }
    }
}

Frequently Asked Questions

Quick answers to common questions

Which platforms are supported?

SecureCrypt supports Windows 10/11, macOS 12+, and major Linux distributions. ARM architectures are supported through Rosetta 2 on Apple Silicon.

Is the source code open-source?

Yes! The core engine is MIT-licensed on GitHub. Enterprise features are available under a commercial license.

How do I generate keys?

Navigate to 'Key Management' > 'Generate Keys' in the app. Keys are stored encrypted using Windows DPAPI or Linux Kernel Keyring.
